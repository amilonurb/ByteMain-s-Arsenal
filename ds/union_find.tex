vector<int> pai(NUM_NODES, NILL);

int _find(int filho) { 
    return (pai[filho] == NILL) ? filho : _find(pai[filho]); 
}
 
void _union(int x, int y) { 
    pai[_find(x)] = _find(y); 
}

bool formaCiclo(int x, int y) { 
    return _find(x) == _find(y);
}
