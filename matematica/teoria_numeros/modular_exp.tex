int fastPower(int x, int n, int MOD) {
    int y = x, result = 1;
    while (n > 0) {
        if (n & 1) result = (result % MOD * y % MOD) % MOD;
        y = (y % MOD * y % MOD) % MOD;
        n >>= 1;
    }
    return result;
}
