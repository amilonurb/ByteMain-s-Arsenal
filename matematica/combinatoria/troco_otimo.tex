// n: total de moedas distintas, v: valor
int minTroco(V(int) moeda, int n, int v) {
	v++;
	V(int) memo(v, INF); 
	memo[0]=0;
    for (int i=0;i<n;i++) memo[moeda[i]]=1;
    for (int i=1;i<v;i++)
    	for (int j=0;j<n;j++)
        	if (i >= coins[j] and memo[i-coins[j]] != INF)
            	memo[i] = min(memo[i], memo[i - moeda[j]] + 1);
    if (memo[v] == INF) memo[v] = -1;
    return memo[v];
}
