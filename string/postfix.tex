bool ehOperando(char x){ 
   return (x >= 'a' && x <= 'z') || 
          (x >= 'A' && x <= 'Z'); 
} 
  
string infixa(string ex){ 
    stack <string> s; 
  
    for(int i = 0; ex[i] != '\0'; i++){ 
        if (ehOperando(ex[i])){ 
           string op(1, ex[i]); 
           s.push(op); 
        }
        else{ 
            string op1 = s.top(); 
            s.pop(); 
            string op2 = s.top(); 
            s.pop(); 
            s.push("(" + op2 + ex[i] + 
                   op1 + ")"); 
        } 
    }
    return s.top(); 
}
