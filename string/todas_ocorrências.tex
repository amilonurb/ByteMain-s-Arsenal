vector<int> todasOcorrencias(string &a, string &b){
	int pos = 0;
	vector<int> ocorrencias;
	
	while(true){
		pos = a.find(b, pos);
		if(pos == string::npos) return ocorrencias;
		ocorrencias.push_back(pos);
		pos+=b.size();		
	}
}
