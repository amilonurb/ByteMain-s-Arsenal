int countLongestSubstring(string &s1, string &s2) {
    int n1 = s1.size(), n2 = s2.size(),max = 0;
    for (int i = 0; i < n1; i++) {
        for (int j = 0; j < n2; j++) {
            if (s1[i] == s2[j]) {
                int c = 0, k = 0;
                for (; (k + i) < n1 || (k + j) < n2; k++) {
                    if (s1[k + i] != s2[k + j]) break;
                    c++;
                }
                if (c > max) max = c;
            }
        }
    }
    return max;
}

int lcss(string X, string Y) { 
    int m = X.length(), n = Y.length();
    int result = 0, atual = 0;
    int len[2][n];
    
    for(int i = 0; i <= m; i++){ 
        for(int j = 0; j <= n; j++){ 
            if(i == 0 || j == 0)
                len[atual][j] = 0; 
            else if(X[i - 1] == Y[j - 1]){ 
                len[atual][j] = len[1 - atual][j - 1] + 1; 
                result = max(result, len[atual][j]); 
            }else
                len[atual][j] = 0; 
        }
        atual = 1 - atual; 
    }
    return result; 
}
